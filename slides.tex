\item Atividade de grupo: \footnotesize{e=($T$,$g$,$m$,$t$)}\\

\item Sessão de aprendizagem:\\
\footnotesize{\textit{tarefa}$_{i}$=(\textit{Entrada}, \textit{atividade}, \textit{Saída})}
\end{itemize}
\end{itemize}

       (distributeDiscussionSession ?goals ?learners)))

\begin{itemize}

\item Desenvolvimento de uma ferramenta de autoria de estratégias de planejamento instrucional.

\item Integração com ambientes virtuais de aprendizagem.
